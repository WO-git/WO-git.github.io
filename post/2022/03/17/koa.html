const Router = require("koa-router") //路由  官方没维护 建议@koa/router
const router = new Router()
const query = require("../utils/db").query //引入封装好的mysql query方法  进行增删改查操作
//引入shop 模块
const shop = require('./shop')
//引入banner 模块
const banner = require("./banner")
//引入登录注册模块
const login = require('./login')
//引入文章列表模块
const article = require('./article')
router.get('/index',async(ctx, next)=>{
    const sql = `select * from user`
    await query(sql).then(async(res)=>{
        ctx.body = {
            code:'200',
            message:'操作成功',
            result:{
                data:res,
                path:ctx.path
            }
        }
    })
    await next()
})

router.get('/home',async(ctx, next)=>{
    ctx.body = 'home Page'
    await next()
})

router.use('/shop',shop.routes(),shop.allowedMethods())
router.use(banner.routes(),banner.allowedMethods())
router.use('/login',login.routes(),login.allowedMethods())
router.use(article.routes(),article.allowedMethods())

module.exports = router

# router login.js

const Router = require('koa-router')
const query = require('../utils/db').query
const bodyparser = require('koa-bodyparser')
const jwt = require('jsonwebtoken')
const login = new Router()

//需要抛出请求回来的数据是用 await 和 Promise
login.use(bodyparser())  // ctx.request.body

login.post('/register', async (ctx, next) => {
    let result = await new Promise((resolve, reject) => {
        ({userName} = ctx.request.body)
        let searchSql = `select * from login where userName='${userName}'`
        return query(searchSql, (err, data) => {
            if (err) throw err;
            resolve(data)
        })
    })
    if (result.length > 0) {
        //验证密码
        //判断密码是否一致
       ({userPassword} = ctx.request.body)
        if (result[0].userPassword == userPassword) {
            ctx.body = {
                message: '登录成功',
                userName: result[0].userName,
                token: result[0].token
            }
        } else {
            ctx.body = {
                message: '账号或密码错误'
            }
        }
    } else {
        //注册  添加数据
        ({ userName,userPassword } = ctx.request.body)
        let token = jwt.sign({ userName, userPassword }, 'secret', { expiresIn: 3600 }) //创建token
        let insertSql = `insert into login (userName,userPassword,token) values ('${userName}','${userPassword}','${token}')`
        let result = await new Promise((resolve, reject) => {
            return query(insertSql, (err, data) => {
                if (err) throw err
                resolve(data)
            })
        })
        if (result.affectedRows > 0) {
            ctx.body = {
                message: '注册成功',
                userName,
                token
            }
        } else {
            ctx.body = {
                message: '注册失败'
            }
        }
        
    }
})
module.exports = login

# utils db.js

const mysql = require("mysql")
const config = require('./config.js')

//连接池
const pool = mysql.createPool({
    host: config.database.HOST,
    user: config.database.USERNAME,
    password: config.database.PASSWORD,
    database: config.database.DATABASE
})

//封装query方法  增删改查操作
let query = (sql, values) => {
    return new Promise((resolve, reject) => {
      pool.getConnection((err, connection) => {
        if (err) {
          reject(err)
        } else {
          connection.query(sql, values, (err, rows) => {
            if (err) {
              reject(err)
            } else {
              connection.release() //连接释放的位置 2个都要有,不然或释放不完全,容易断
              resolve(rows)
            }
          })
          connection.release()
        }
      })
    })
  
  }

  exports.query = query

# app.js

const Koa = require("koa")
const static = require("koa-static") //静态资源路径
const path = require("path") 
const mount = require("koa-mount") //静态访问路径前缀
const compose = require("koa-compose") //合并中间件
const cors = require('koa-cors')  //处理跨域
const json = require("koa-json")  //处理json格式
const bodyparser = require("koa-bodyparser") //post 提交的数据 ctx.request.body
const logger = require("koa-logger")  //打印日志
const router = require('./router')
const app = new Koa()
const PORT = 3001

app.use(cors())  //处理跨域
app.use(json())
app.use(logger())
app.use(bodyparser({
  enableTypes: ['json', 'form', 'text']
}))
app.use(mount('/tx',static(path.join(__dirname,'./public')))) //静态资源
app.use(router.routes(),router.allowedMethods()) //挂在路由
app.on('error', err => {  //错误处理
    console.error('server error', err)
  });
app.listen(PORT,()=>{
    console.log(`Running start ${PORT}`)
}).setTimeout(5*60*1000)
