      pool.getConnection((err, connection) => {
        if (err) {
          reject(err)
        } else {
          connection.query(sql, values, (err, rows) => {
            if (err) {
              reject(err)

              connection.release() //è¿æ¥é‡Šæ”¾çš„ä½ç½® 2ä¸ªéƒ½è¦æœ‰,ä¸ç„¶æˆ–é‡Šæ”¾ä¸å®Œå…¨,å®¹æ˜“æ–­
